def count_letters(lexicon, msg):
    for letter in msg:
        if letter in lexicon.keys():
            lexicon[letter] += 1
    return lexicon

lex = {'w':0, '1':0, 'c':0}
case2 = count_letters(lex,"hello world")
print(case2)
vowels = {'a':0, 'o':0, 'e':0, 'i':0, 'u':0}
case1 = count_letters(vowels,"Why did I choose CS?")
print(case1)
