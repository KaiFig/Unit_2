def get_l3tt3r(msg):
    output = ""
    for i in msg:
        if i == "a" or i == "A":
            output += "4"
        elif i == "e" or i == "E":
            output += "3"
        elif i == "i" or i == "I":
            output += "1"
        elif i == "o" or i == "O":
            output += "0"
        elif i == " ":
            output += "_"
        else:
            output += i
    return output

test1 = get_l3tt3r("hello world")
print(test1)

test2 = get_l3tt3r("Why did I choose CS?")
print(test2)

test3 = get_l3tt3r("Remember the Figure Caption")
