from matplotlib import pyplot as plt

def function():
    x = []
    y = []
    n = 100
    m = -10
    for i in range(n):
        mstr = str(m)
        y.append(mstr)
        yfunc = ((2*(m+5))**2)
        x.append(yfunc)
        m += 0.2

    return x,y

data_y, data_x = function()
plt.plot(data_x, data_y, color="red")
plt.xlabel("variable x")
plt.ylabel("$y = 2(x+5)^2}$")
plt.show()
